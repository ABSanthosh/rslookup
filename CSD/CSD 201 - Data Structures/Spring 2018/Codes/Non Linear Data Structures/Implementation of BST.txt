//Implementation of BST
#include<stdio.h>
typedef struct tree{
int data;
struct tree *left;
struct tree *right;
}tree;
tree *root=NULL;
tree *p=NULL;
tree *insert(tree *,int);
void main()
{
int n,ch;
while(1)
{
printf("1. Insert\n");
printf("2. Delete\n");
printf("3. Display Inorder\n");
printf("4. Exit\n");
printf("Enter your choice\n");
scanf("%d",&ch);
switch(ch)
{
case 1: printf("Enter element to be inserted\n");
scanf("%d",&n);
root=insert(root,n);
break;
case 2: delete();
break;
case 3: display(root);
break;
case 4: exit(0);
}
}
}
tree *insert(tree *temp,int n)
{
if(temp==NULL)
{
tree *new=(tree *)malloc(sizeof(tree));
new->data=n;
new->left=NULL;
new->right=NULL;
return new;
}
if(n<temp->data)
temp->left=insert(temp->left,n);
else 
temp->right=insert(temp->right,n);
return temp;
}
void delete()
{
int n;
if(root==NULL)
{
printf("Tree is empty\n");
return;
}
printf("Enter element to be deleted\n");
scanf("%d",&n);
p=root;
search(root,n);
}

void search(tree *x,int n)
{
if(x==NULL)
{printf("Element not present \n");
return;
}
else if(n<x->data)
{
p=x;
search(x->left,n);
}
else if(n>x->data)
{
p=x;
search(x->right,n);
}
else if(n==x->data)
deletenode(x);
}
void deletenode(tree *x)
{
int successor;
//x i s a leaf node
if((x->left==NULL) &&(x->right==NULL))
{
if(p==x)
root=NULL;
else if(x==p->left)
p->left=NULL;
else 
p->right=NULL;
free(x);
}
//if x has only left child
else if(x->right==NULL)
{
if(p==x)
{
root=x->left;
p=root;
}
else if(x==p->left)
p->left=x->left;
else
p->right=x->left;
free(x);
}
//if x has only right child
else if(x->left==NULL)
{
if(p==x)
{
root=x->right;
p=root;
}
else if(x==p->left)
p->left=x->right;
else
p->right=x->right;
free(x);
}
//if x has both children
else 
{
successor=smallest(x->right);
search(root,successor);
x->data=successor;
}
}

int smallest(tree *x)
{
if(x->left!=NULL)
smallest(x->left);
else
return x->data;
}
void display(tree *root)
{
if(root==NULL)
return;
display(root->left);
printf("%d\n",root->data);
display(root->right);
}
