#include<stdio.h>
int arr[20];
void main()
{
  		int n,i;
		printf("Enter the size of array\n");
  		scanf("%d", &n);
  		printf("Enter the elements:");
  		for(i=0; i<n; i++)
    		scanf("%d", &arr[i]);
  		partition(arr, 0, n-1);
  		printf("Sorted array:");
  		for(i=0; i<n; i++)
    		printf("%d\n", arr[i]);
}
void partition(int arr[], int p, int r)
{
  		int q;
  		if(p<r)
		{
    			q=(p+r)/2;
    			partition(arr, p, q);
    			partition(arr, q+1, r);
    			merge(arr, p, q, r);
  		}
}
void merge(int arr[], int p, int q, int r)
{
  		int left[10], right[10];
  		int n1, n2, i, j, k;
  		n1 = q-p+1;
  		n2 = r-q;
  		for(i=0; i<n1; i++)
    		left[i] = arr[p+i];
  		for(j=0; j<n2; j++)
    		right[j] = arr[q+j+1];
  		left[i] = 9999;
  		right[j] = 9999;
  		i = 0;
  		j = 0;
  		for(k=p; k<=r; k++)
		{
    			if(left[i] <= right[j])
      				arr[k] = left[i++];
    			else
      				arr[k] = right[j++];
  		}
}
