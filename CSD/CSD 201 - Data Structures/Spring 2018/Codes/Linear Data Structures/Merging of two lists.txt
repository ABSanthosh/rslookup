typedef struct node{

}node1;

	break;

void create1()
{
int n;
printf("Enter the element\n");
scanf("%d",&n);
if(head1==NULL)
{
	head1=(node1 *)malloc(sizeof(node1));
	if(head1==NULL)
	{
		printf("memory not allocated\n");
		exit(0);
	}
	head1->data=n;
	head1->next=NULL;
}
else
{
	node1 *ptr=(node1 *)malloc(sizeof(node1));
	if(ptr==NULL)
	{
		printf("memory not allocated\n");
		exit(0);
	}
	ptr->data=n;
	ptr->next=NULL;
	node1 *temp=head1;
	while(temp->next!=NULL)
		temp=temp->next;
	temp->next=ptr;
}
}

   while(t->next!=NULL)
