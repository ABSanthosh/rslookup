#include<stdio.h>
#include<stdlib.h>

typedef struct bstnode{
int data;
struct  bstnode *left;
struct  bstnode *right;
}node;

node *insert(node *, int);
void inorder(node *);
int height(node *);

void main(){
int ch,element,h;
node *root=NULL;

while(1){
printf("\n1. To insert element in BST\n");
printf("2. To print inorder of BST\n");
printf("3. To find height of BST\n");
printf("4. Exit\n");
printf("Enter your choice\n");
scanf("%d", &ch);
switch(ch){

case 1: printf("Enter an element to insert\n");
        scanf("%d",&element);
        root=insert(root,element);
        break;
case 2: inorder(root);
        break;
case 3: h=height(root);
        printf("Height of tree is %d",h);
        break;
case 4:
        exit(0);
default:  printf("You have entered wrong choice");
          break;
}
}
}

node *insert(node *root,int element)
{
    if(root==NULL)
    {
        node *temp=(node *)malloc(sizeof(node));
        temp->data=element;
        temp->left=temp->right=NULL;
        return temp;
    }
    else if(element<root->data)
        root->left=insert(root->left,element);
    else
        root->right=insert(root->right,element);
    return root;
}

void inorder(node *rootnode)
{
    if(rootnode == NULL)
       return;
    if(rootnode->left != NULL)
        inorder(rootnode->left);
    printf("%d\n", rootnode->data);
    if(rootnode->right!= NULL)
        inorder(rootnode->right);
}

int height(node *root)
{
	if(root==NULL)
    		return 0;
	int ld=height(root->left);
	int rd=height(root->right);
	if(ld>rd)
    		return ld+1;
	else
    		return rd+1;
}
